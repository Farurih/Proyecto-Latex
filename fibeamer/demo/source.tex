    \framesubtitle{All integers divide zero}
    \begin{definition}
      $\forall a,b\in\mathds{Z}: a\mid b\iff\exists c\in\mathds{Z}:a\cdot c=b$ 
    \end{definition}
    \begin{theorem}
      $\forall a\in\mathds{Z}: a\mid 0$
    \end{theorem}
    \begin{proof}[Proof\nopunct]
      $\forall a\in\mathds{Z}: a\cdot 0=0$
    \end{proof}
